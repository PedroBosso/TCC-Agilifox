interface FotoPreviewProps {
  fotoBase64?: string | null;
  width?: number;
  height?: number;
  borderRadius?: number;
  showPlaceholder?: boolean;
}

export function FotoPreview({
  fotoBase64,
  width = 120,
  height = 120,
  borderRadius = 60,
  showPlaceholder = true,
}: FotoPreviewProps) {
  if (fotoBase64) {
    return (
      <Image
        source={{ uri: `data:image/jpeg;base64,${fotoBase64}` }}
        style={{
          width,
          height,
          borderRadius,
          borderWidth: 3,
          borderColor: '#e49c15',
        }}
      />
    );
  }

  if (!showPlaceholder) return null;

  return (
    <View
      style={{
        width,
        height,
        borderRadius,
        backgroundColor: '#e8dcc8',
        alignItems: 'center',
        justifyContent: 'center',
        borderWidth: 2,
        borderColor: '#d4c4a8',
        borderStyle: 'dashed',
      }}
    >
      <Text style={{ fontSize: 40 }}>👤</Text>
      <Text style={{ fontSize: 12, color: '#999', marginTop: 4 }}>
        Sem foto
      </Text>
    </View>
  );
}
